fig1 = plt.figure()
ax1 = fig1.add_subplot(121, projection='3d')
ax2 = fig1.add_subplot(122, projection='3d')
ax2.set_aspect("equal") # en/boy/yukseklik oranini esit yapiyoruz!

# Dikkat! Olculer sizi yaniltabilir
# en/boy/yukseklik degerlerine dikkat edin

fig2 = plt.figure()
ax3 = fig2.add_subplot(121, projection='3d')
ax3.set_aspect("equal")


def fxy(x, y):
    f = x**2 + y**2
    return -f

x = np.arange(-10,10)
y = np.arange(-10,10)
# X, Y = np.mgrid[-10:10, -10:10]
X, Y = np.meshgrid(x, y) #(x,y) ikililerini olusturuyoruz. 
# X,Y matrislerinin karsilikli her degeri bir (x,y)


# (x,x) ikilillerine karsilik z degerlerini bulacagiz
Z = fxy(X,Y)
# X,Y,Z matrislerinin karsilikli her degeri bir (x,y,z)

# (x,y,z) lerden olusan bir ag olusturuyoruz.
ax1.plot_wireframe(X, Y, Z, color="r", rstride=5, cstride=5)
ax2.plot_wireframe(X, Y, Z, color="b", rstride=1, cstride=2)

# Bu ise tam bir yuzey goruntusu
from matplotlib import cm
ax3.plot_surface(X, Y, Z, rstride=1, cstride=1, cmap=cm.coolwarm)

plt.show()
